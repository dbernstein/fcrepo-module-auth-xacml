46      /**
47       * Extract a policy set or policy ID for the document.
48       *
49       * @param policyStream the policy input
50       * @return an identifier
51       */
52      public static String getID(final InputStream policyStream) {
53          try {
54              // create the factory
55              final DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
56              factory.setIgnoringComments(true);
57  
58              factory.setNamespaceAware(true);
59              factory.setValidating(false);
60              final DocumentBuilder db = factory.newDocumentBuilder();
61  
62              // Parse the policy content
63              final Document doc = db.parse(policyStream);
64  
65              final String result = getID(doc);
66              if (result == null) {
67                  throw new Error("Cannot find policy ID");
68              }
69              return result;
70          } catch (final Exception e) {
71              throw new Error("Unable to parse policy", e);
72          }
73      }
74  
75      /**
76       * Get the ID of the XACML policy document.
77       *
78       * @param doc the DOM
79       * @return the ID
80       */
81      public static String getID(final Document doc) {
82          // handle the policy, if it's a known type
83          final Element root = doc.getDocumentElement();
84          final String name = root.getTagName();
85          if (name.equals("Policy")) {
86              return root.getAttribute("PolicyId");
87          } else if (name.equals("PolicySet")) {
88              return root.getAttribute("PolicySetId");
89          } else {
90              return null;
91          }
92      }

104     /**
105      * Find the nearest real Modeshape node for a given Modeshape path.
106      *
107      * @param modepath the path in ModeShape
108      * @param session a session
109      * @return a Node in session
110      */
111     public static Node getFirstRealNode(final String modepath, final Session session) {
112         Node node = null;
113         for (String path = modepath; path.contains("/{"); path = path.substring(0, path.lastIndexOf("/{"))) {
114             try {
115                 node = session.getNode(path);
116                 break;
117             } catch (final PathNotFoundException expected) {
118             } catch (final RepositoryException e) {
119                 throw new Error("Cannot reach repository", e);
120             }
121         }
122         if (node == null) {
123             try {
124                 node = session.getRootNode();
125             } catch (final RepositoryException e) {
126                 throw new Error("Cannot reach repository", e);
127             }
128         }
129         return node;
130     }

132     /**
133      * Get the action ids.
134      *
135      * @param context the evaluation context
136      * @return a set of actions
137      */
138     public static Set<String> getActions(final EvaluationCtx context) {
139         final Set<String> result = new HashSet<String>();
140         final EvaluationResult eval =
141                 context.getActionAttribute(URI.create("http://www.w3.org/2001/XMLSchema#string"),
142                         URIConstants.ATTRIBUTEID_ACTION_ID, null);
143         if (eval == null) {
144             return null;
145         }
146         if (eval.getStatus() == null) {
147             final AttributeValue val = eval.getAttributeValue();
148             if (val != null && val.getValue() != null) {
149                 result.add(val.getValue().toString());
150             }
151         }
152         return result;
153     }
