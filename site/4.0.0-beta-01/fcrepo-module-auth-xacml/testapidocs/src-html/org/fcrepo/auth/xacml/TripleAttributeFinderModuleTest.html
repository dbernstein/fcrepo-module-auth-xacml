160    @Test
161    public void testFindAttribute() throws RepositoryException {
162
163        final String resourceId = "/{ns}path/{ns}to/{ns}resource";
164
165        when(mockNodeService.getObject(mockSession, resourceId)).thenReturn(mockFedoraResource);
166        when(mockFedoraResource.getTriples(any(IdentifierTranslator.class))).thenReturn(mockRdfStream);
167        when(mockIdentifierTranslator.getSubject(any(String.class))).thenReturn(mockResource);
168        when(mockFedoraResource.getPath()).thenReturn(resourceId);
169        when(mockRdfStream.asModel()).thenReturn(mockModel);
170        when(mockModel.listObjectsOfProperty(any(Resource.class),
171                any(Property.class))).thenReturn(mockMatches);
172        when(mockMatches.hasNext()).thenReturn(true).thenReturn(false);
173        when(mockMatches.next()).thenReturn(mockRDFNode);
174        when(mockRDFNode.asResource()).thenReturn(mockResource);
175        when(mockResource.getURI()).thenReturn("SamIAm");
176
177        final EvaluationResult result = doFindAttribute(resourceId);
178
179        final AttributeValue attributeValue = result.getAttributeValue();
180        assertNotNull("Evaluation.attributeValue shoud not be null!", attributeValue);
181        assertTrue("Evaluation.attributeValue should be a bag!", attributeValue.isBag());
182
183        final URI value = (URI) attributeValue.getValue();
184        assertNotNull("EvaluationResult value should not be null!", value);
185        assertEquals(value.toString(), "SamIAm");
186    }

323    private void assertIsEmptyResult(final EvaluationResult result) {
324        final BagAttribute attributeValue = (BagAttribute) result.getAttributeValue();
325        assertNotNull("Evaluation.attributeValue shoud not be null!", attributeValue);
326        assertTrue("Evaluation.attributeValue should be a bag!", attributeValue.isBag());
327
328        assertTrue("Attribute bag should be empty!", attributeValue.isEmpty());
329        final Object value = attributeValue.getValue();
330        assertNull("EvaluationResult value should be null!", value);
331    }
332
333    private EvaluationResult doFindAttribute(final String resourceId) {
334        return doFindAttribute(-1, resourceId, null);
335    }
336
337    private EvaluationResult doFindAttribute(final String resourceId, final String[] actions) {
338        return doFindAttribute(-1, resourceId, actions);
339    }
340
341    private EvaluationResult doFindAttribute(final int argDesignatorType, final String resourceId,
342            final String[] actions) {
343        final URI attributeType = URI.create("http://www.w3.org/2001/XMLSchema#anyURI");
344        final URI attributeId = URI.create("uri:att-id");
345        final URI issuer = null;
346        final URI subjectCategory = null;
347        final EvaluationCtx context = evaluationCtx(resourceId, actions);
348        final int designatorType = argDesignatorType == -1 ? RESOURCE_TARGET : argDesignatorType;
349
350        final EvaluationResult result = finder.findAttribute(attributeType,
351                                                             attributeId,
352                                                             issuer,
353                                                             subjectCategory,
354                                                             context,
355                                                             designatorType);
356
357        assertNotNull("EvaluationResult should not be null!", result);
358        return result;
359    }
360
361    private EvaluationCtx evaluationCtx(final String resourceId, final String[] actions) {
362        final FedoraEvaluationCtxBuilder builder = new FedoraEvaluationCtxBuilder();
363        if (resourceId != null) {
364            builder.addResourceID(resourceId);
365        }
366
367        if (actions != null && actions.length > 0) {
368            builder.addActions(actions);
369        }
370        builder.addSubject("user", null);
371        return builder.build();
372    }
