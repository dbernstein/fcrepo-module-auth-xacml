016package org.fcrepo.auth.xacml;
017
018import static javax.ws.rs.core.Response.Status.NO_CONTENT;
019import static org.junit.Assert.assertEquals;
020
021import java.io.ByteArrayInputStream;
022import java.io.File;
023import java.text.MessageFormat;
024
025import org.apache.commons.io.FileUtils;
026import org.apache.commons.io.FilenameUtils;
027import org.apache.http.HttpResponse;
028import org.apache.http.client.HttpClient;
029import org.apache.http.client.methods.HttpPatch;
030import org.apache.http.entity.BasicHttpEntity;
031import org.fcrepo.auth.roles.common.integration.RolesFadTestObjectBean;
032import org.slf4j.Logger;
033import org.slf4j.LoggerFactory;
034
035/**
036 * Some XACML loading and linking utilities for tests.
037 *
038 * @author Gregory Jansen
039 */
040public class XACMLTestUtil {
041
042    /**
043     * Class logger.
044     */
045    private static final Logger LOGGER = LoggerFactory.getLogger(XACMLTestUtil.class);
046
047    private XACMLTestUtil () {
048        // not called
049    }
050
051    public static void addPolicy(final RolesFadTestObjectBean policies, final File path) throws Exception {
052        final String xacml = FileUtils.readFileToString(path);
053        policies.addDatastream(FilenameUtils.getBaseName(path.getName()), xacml);
054    }
055
056    /**
057     * Removes the policy link from an object.
058     *
059     * @param client
060     * @param serverAddress
061     * @param objectPath
062     * @throws Exception
063     */
064    public static void unlinkPolicies(final HttpClient client, final String serverAddress, final String objectPath)
065            throws Exception {
066        final String subjectURI = serverAddress + objectPath;
067        final HttpPatch patch = new HttpPatch(subjectURI);
068        // setAuth(patch, "fedoraAdmin");
069        patch.addHeader("Content-Type", "application/sparql-update");
070        final BasicHttpEntity e = new BasicHttpEntity();
071        e.setContent(new ByteArrayInputStream(
072                ("DELETE { <" + subjectURI + "> <http://fedora.info/definitions/v4/authorization#policy> ?o } " +
073                        "WHERE { <" + subjectURI + "> <http://fedora.info/definitions/v4/authorization#policy> ?o }")
074                        .getBytes()));
075        patch.setEntity(e);
076        LOGGER.debug("PATCH: {}", patch.getURI());
077        final HttpResponse response = client.execute(patch);
078        assertEquals(NO_CONTENT.getStatusCode(), response.getStatusLine().getStatusCode());
079    }
080
081    /**
082     * Links an object to a policy(set).
083     *
084     * @param client
085     * @param serverAddress
086     * @param objectPath
087     * @param policyPath
088     * @throws Exception
089     */
090    public static void linkPolicy(final HttpClient client, final String serverAddress, final String objectPath,
091            final String policyPath) throws Exception {
092        final String subjectURI = serverAddress + objectPath;
093        final String policyURI = policyPath;
094        final HttpPatch patch = new HttpPatch(subjectURI);
095        // setAuth(patch, "fedoraAdmin");
096        patch.addHeader("Content-Type", "application/sparql-update");
097        final BasicHttpEntity e = new BasicHttpEntity();
098        final StringBuilder sb = new StringBuilder();
099        sb.append("INSERT { ")
100                .append(MessageFormat
101                        .format("<{0}> <http://www.w3.org/1999/02/22-rdf-syntax-ns#type> " +
102                                "<http://fedora.info/definitions/v4/authorization#xacmlAssignable> . ",
103                                subjectURI)).append(
104                        MessageFormat.format("<{0}> <http://fedora.info/definitions/v4/authorization#policy> <{1}> . ",
105                                subjectURI, policyURI)).append("} WHERE { }");
106        e.setContent(new ByteArrayInputStream(sb.toString().getBytes()));
107        patch.setEntity(e);
108        LOGGER.debug("PATCH: {}", patch.getURI());
109        final HttpResponse response = client.execute(patch);
110        assertEquals(NO_CONTENT.getStatusCode(), response.getStatusLine().getStatusCode());
111    }
112}
