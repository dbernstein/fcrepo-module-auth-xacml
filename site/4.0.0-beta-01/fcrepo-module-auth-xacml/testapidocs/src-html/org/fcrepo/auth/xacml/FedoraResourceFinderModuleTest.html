016package org.fcrepo.auth.xacml;
017
018import static org.junit.Assert.assertFalse;
019import static org.junit.Assert.assertTrue;
020import static org.mockito.Matchers.anyString;
021import static org.mockito.Mockito.when;
022import static org.mockito.MockitoAnnotations.initMocks;
023
024import org.junit.After;
025import org.junit.Before;
026import org.junit.Test;
027import org.mockito.Mock;
028
029import java.util.Set;
030
031import javax.jcr.Node;
032import javax.jcr.NodeIterator;
033import javax.jcr.Session;
034import javax.jcr.nodetype.NodeType;
035
036import org.fcrepo.http.commons.session.SessionFactory;
037
038import org.jboss.security.xacml.sunxacml.attr.AttributeValue;
039import org.jboss.security.xacml.sunxacml.finder.ResourceFinderResult;
040
041
042/**
043 * @author Andrew Woods
044 *         Date: 5/9/14
045 * @author Esme Cowles
046 */
047public class FedoraResourceFinderModuleTest {
048    FedoraResourceFinderModule resourceFinder;
049
050    @Mock SessionFactory mockSessionFactory;
051    @Mock Session mockSession;
052    @Mock AttributeValue mockParent;
053    @Mock Node mockParentNode;
054    @Mock NodeType mockNodeType;
055    @Mock Node mockChildNode;
056    @Mock Node mockGrandchildNode;
057    @Mock NodeIterator mockParentIterator;
058    @Mock NodeIterator mockChildIterator;
059    @Mock NodeIterator mockGrandchildIterator;
060
061    @Before
062    public void setUp() throws Exception {
063        initMocks(this);
064        resourceFinder = new FedoraResourceFinderModule();
065        resourceFinder.sessionFactory = mockSessionFactory;
066
067        when( mockSessionFactory.getInternalSession() ).thenReturn(mockSession);
068        when( mockParent.getValue() ).thenReturn("/foo");
069        when( mockSession.getNode("/foo") ).thenReturn(mockParentNode);
070
071        when( mockParentNode.getNodes() ).thenReturn(mockParentIterator);
072        when( mockParentIterator.hasNext() ).thenReturn(true,false);
073        when( mockParentIterator.nextNode() ).thenReturn(mockChildNode);
074
075        when( mockChildNode.getPath() ).thenReturn("/foo/bar");
076        when( mockChildNode.getName() ).thenReturn("bar");
077        when( mockChildNode.getPrimaryNodeType() ).thenReturn(mockNodeType);
078        when( mockNodeType.isNodeType(anyString()) ).thenReturn(false);
079        when( mockChildNode.getNodes() ).thenReturn(mockChildIterator);
080        when( mockChildIterator.hasNext() ).thenReturn(true,false);
081        when( mockChildIterator.nextNode() ).thenReturn(mockGrandchildNode);
082
083        when( mockGrandchildNode.getPath() ).thenReturn("/foo/bar/baz");
084        when( mockGrandchildNode.getName() ).thenReturn("baz");
085        when( mockGrandchildNode.getPrimaryNodeType() ).thenReturn(mockNodeType);
086        when( mockGrandchildNode.getNodes() ).thenReturn(mockGrandchildIterator);
087        when( mockGrandchildIterator.hasNext() ).thenReturn(false);
088        when( mockGrandchildIterator.nextNode() ).thenReturn(null);
089
090    }
091
092    @After
093    public void tearDown() throws Exception {
094
095    }
096
097    @Test
098    public void testIsChildSupported() throws Exception {
099        assertTrue( resourceFinder.isChildSupported() );
100    }
101
102    @Test
103    public void testIsDescendantSupported() throws Exception {
104        assertTrue( resourceFinder.isDescendantSupported() );
105    }
106
107    @Test
108    public void testFindChildResources() throws Exception {
109        final ResourceFinderResult result = resourceFinder.findChildResources( mockParent, null );
110        final Set resources = result.getResources();
111        assertTrue( "Child not found", resources.contains("/foo/bar") );
112        assertFalse( "Grandchild should not be found", resources.contains("/foo/bar/baz") );
113    }
114
115    @Test
116    public void testFindDescendantResources() throws Exception {
117        final ResourceFinderResult result = resourceFinder.findDescendantResources( mockParent, null );
118        final Set resources = result.getResources();
119        assertTrue( "Child not found", resources.contains("/foo/bar") );
120        assertTrue( "Grandchild not found", resources.contains("/foo/bar/baz") );
121    }
122}
