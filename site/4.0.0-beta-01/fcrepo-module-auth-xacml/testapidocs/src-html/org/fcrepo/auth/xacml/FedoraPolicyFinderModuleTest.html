131    @Test
132    public void testFindPolicyOnTargetNode() throws Exception {
133
134        when(mockNode.hasProperty(eq(XACML_POLICY_PROPERTY))).thenReturn(true);
135        when(mockNode.getProperty(eq(XACML_POLICY_PROPERTY))).thenReturn(mockPolicyProperty);
136
137        when(mockPolicyDs.getContent()).thenReturn(this.getClass().getResourceAsStream("/xacml/testPolicy.xml"));
138
139        final FedoraEvaluationCtxBuilder ctxBuilder = new FedoraEvaluationCtxBuilder();
140        ctxBuilder.addResourceID("/{}myPath");
141        final Set<String> subjectSet = new HashSet<>();
142        ctxBuilder.addSubject("test", subjectSet);
143
144        final EvaluationCtx ctx = ctxBuilder.build();
145
146        final PolicyFinderResult result = finderModule.findPolicy(ctx);
147
148        assertFalse(result.notApplicable());
149        assertFalse(result.indeterminate());
150        assertNotNull(result.getPolicy());
151    }

153    @Test
154    public void testFindPolicyByIdReference() throws Exception {
155        final String policyPath = "/path/to/policy";
156        final String idPath = POLICY_URI_PREFIX + policyPath;
157        final URI idReference = new URI(idPath);
158
159        when(mockPolicyDs.getContent()).thenReturn(this.getClass().getResourceAsStream("/xacml/testPolicy.xml"));
160        when(mockDsService.getDatastream(any(Session.class), eq(policyPath))).thenReturn(mockPolicyDs);
161
162        final PolicyFinderResult result = finderModule.findPolicy(idReference, 0, null, null);
163
164        assertFalse(result.notApplicable());
165        assertFalse(result.indeterminate());
166        assertNotNull(result.getPolicy());
167    }

169    @Test
170    public void testFindPolicySet() throws Exception {
171
172        when(mockNode.hasProperty(eq(XACML_POLICY_PROPERTY))).thenReturn(true);
173        when(mockNode.getProperty(eq(XACML_POLICY_PROPERTY))).thenReturn(mockPolicyProperty);
174
175        when(mockPolicyDs.getContent())
176        .thenReturn(this.getClass().getResourceAsStream("/xacml/adminRolePolicySet.xml"));
177
178        final String referencedId = "fcrepo:policies/AdminPermissionPolicySet";
179        final Datastream referencedPolicyDs = mock(Datastream.class);
180        when(referencedPolicyDs.getContent()).thenReturn(
181                this.getClass().getResourceAsStream("/xacml/adminPermissionPolicySet.xml"));
182        when(mockDsService.getDatastream(any(Session.class), eq(referencedId))).thenReturn(referencedPolicyDs);
183
184        final FedoraEvaluationCtxBuilder ctxBuilder = new FedoraEvaluationCtxBuilder();
185        ctxBuilder.addResourceID("/{}myPath");
186        final Set<String> subjectSet = new HashSet<>();
187        subjectSet.add("admin");
188        ctxBuilder.addSubject("username", subjectSet);
189
190        final EvaluationCtx ctx = ctxBuilder.build();
191
192        final PolicyFinderResult result = finderModule.findPolicy(ctx);
193
194        assertFalse(result.notApplicable());
195        assertFalse(result.indeterminate());
196        assertNotNull(result.getPolicy());
197
198        final List<?> policyChildren = result.getPolicy().getChildElements();
199        assertEquals("Combiner algorithm child not found", 1, policyChildren.size());
200        final PolicyCombinerElement combiner = (PolicyCombinerElement) policyChildren.get(0);
201        final PolicyReference policyRef = (PolicyReference) combiner.getElement();
202        assertEquals("URI for policy reference was not the expected value", "fcrepo:policies/AdminPermissionPolicySet",
203                policyRef.getReference().toString());
204    }
